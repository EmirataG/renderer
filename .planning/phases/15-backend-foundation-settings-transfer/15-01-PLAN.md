Create `src/shared/exportSettings.ts`:
- Import { Type, Static } from '@sinclair/typebox'
- Define ScoreRegionSchema: Type.Object({ x: Type.Number(), y: Type.Number(), width: Type.Number(), height: Type.Number() })
- Define BorderStyleSchema: Type.Union of literals: 'none', 'line', 'double-line', 'ornate-1', 'ornate-2', 'flourish'
- Define MusicFontSchema: Type.Union of literals: 'Bravura', 'Petaluma', 'Leland', 'Gootville', 'Leipzig'
- Define ExportSettingsSchema: Type.Object with ALL 17 fields from App.tsx:
  - fps: Type.Number({ minimum: 15, maximum: 60 })
  - scoreColor: Type.String({ pattern: '^#[0-9a-fA-F]{6}$' })
  - scoreShadowDistance: Type.Number({ minimum: 0, maximum: 6 })
  - hideUnplayedNotes: Type.Boolean()
  - smoothReveal: Type.Boolean()
  - scoreRegion: Type.Union([ScoreRegionSchema, Type.Null()])
  - scoreBorder: BorderStyleSchema
  - scoreScale: Type.Number({ minimum: 0.5, maximum: 1.5 })
  - musicFont: MusicFontSchema
  - activeNoteheadColor: Type.Union([Type.String({ pattern: '^#[0-9a-fA-F]{6}$' }), Type.Null()])
  - activeNoteheadScale: Type.Number({ minimum: 1, maximum: 1.6 })
  - activeNoteheadEntryMs: Type.Number({ minimum: 0, maximum: 300 })
  - activeNoteheadHoldMs: Type.Number({ minimum: 0, maximum: 1000 })
  - activeNoteheadExitMs: Type.Number({ minimum: 0, maximum: 1000 })
  - colorFullNote: Type.Boolean()
  - audioDuration: Type.Optional(Type.Number({ minimum: 0 }))
- Define SyncAnchorsSchema: Type.Record(Type.String(), Type.Number()) -- represents Object.fromEntries(Map)
- Export types: type ExportSettings = Static<typeof ExportSettingsSchema>
- Export types: type SyncAnchors = Static<typeof SyncAnchorsSchema>

Create `src/shared/validation.ts`:
- Import { Value } from '@sinclair/typebox/value'
- Import ExportSettingsSchema from exportSettings
- validateExportSettings(settings: unknown): string[] -- uses Value.Check(ExportSettingsSchema, settings), returns error paths and messages from Value.Errors
- validateSyncAnchors(anchors: unknown): string[] -- checks: is object, not null, not empty (explicit error: "syncAnchors is empty -- ensure Map is serialized with Object.fromEntries()"), each value is a non-negative number
